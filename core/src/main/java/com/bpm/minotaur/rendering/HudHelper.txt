// Helper method for Hud.java
    private String scrambleOf(String text) {
        if (text == null || text.isEmpty()) return text;
        // Cheap scramble: just swap pairs based on random logic or time if possible, 
        // but here we just want it to look "wrong" without being illegible?
        // Actually user said "string tetxt gets jumbled".
        
        // Simple random shift of chars? 
        // We need it to be consistent per frame? No, jumbled suggests changing.
        // Let's swap some characters.
        char[] chars = text.toCharArray();
        if (chars.length < 2) return text;

        if (com.badlogic.gdx.math.MathUtils.randomBoolean(0.3f)) { // 30% chance to scramble per frame call? 
           // That might flicker too much. 
           // Better to use a deterministic scramble based on time? 
           // Hud doesn't have 'time' easily accessible in this Helper without passing it.
           // Let's use MathUtils.random() and hope 60FPS flickering isn't too seizure inducing.
           // Or just swap stable indices.
           int idx1 = com.badlogic.gdx.math.MathUtils.random(chars.length - 1);
           int idx2 = com.badlogic.gdx.math.MathUtils.random(chars.length - 1);
           char temp = chars[idx1];
           chars[idx1] = chars[idx2];
           chars[idx2] = temp;
        }
        return new String(chars);
    }
